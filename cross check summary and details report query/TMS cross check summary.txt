				count(distinct (case when cco.checked_by_tms_user_id is not null then bs.outlet_id else null end)) totalOutletCount,

                cco.checked_by_tms_user_id is not null
				group   by cco.checked_by_tms_user_id
			)ots ON ots.user_id=vta.user_id

                and     vta.teritory_id like :teritoryId || '%'
